# –í–æ–∑–º–æ–∂–Ω—ã–µ —Å–∏–º–≤–æ–ª—ã

# –§—É–Ω–∫—Ü–∏—è –¥–ª—è –≥–µ–Ω–µ—Ä–∞—Ü–∏–∏ —Å–ª—É—á–∞–π–Ω—ã—Ö —Å–∏–º–≤–æ–ª–æ–≤

def check_win(symbol1, symbol2, symbol3):
    if symbol1 == symbol2 == symbol3:
        if symbol1 == "üçí":
            return "–í—ã –≤—ã–∏–≥—Ä–∞–ª–∏ 5x —Å—Ç–∞–≤–∫—É!", 5
        elif symbol1 == "üîî":
            return "–í—ã –≤—ã–∏–≥—Ä–∞–ª–∏ 3x —Å—Ç–∞–≤–∫—É!", 3
        else:
            return "–í—ã –≤—ã–∏–≥—Ä–∞–ª–∏ 2x —Å—Ç–∞–≤–∫—É!", 2

        return "–ü–æ–ø—Ä–æ–±—É–π—Ç–µ –µ—â–µ —Ä–∞–∑.", 0
