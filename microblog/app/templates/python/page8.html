def load_maps_from_file(filename):
    maps = []  # –°–ø–∏—Å–æ–∫ –¥–ª—è —Ö—Ä–∞–Ω–µ–Ω–∏—è –≤—Å–µ—Ö –∫–∞—Ä—Ç

    with open(filename, 'r', encoding='utf-8') as file:
        for line in file:
            line = line.strip()  # –£–¥–∞–ª—è–µ–º –ø—Ä–æ–±–µ–ª—ã –∏ —Å–∏–º–≤–æ–ª –Ω–æ–≤–æ–π —Å—Ç—Ä–æ–∫–∏
            if len(line) == 0 and current_map:
                maps.append(current_map)  # –î–æ–±–∞–≤–ª—è–µ–º —Ç–µ–∫—É—â—É—é –∫–∞—Ä—Ç—É –≤ —Å–ø–∏—Å–æ–∫ –∫–∞—Ä—Ç
                current_map = []  # –û—á–∏—â–∞–µ–º –≤—Ä–µ–º–µ–Ω–Ω—ã–π —Å–ø–∏—Å–æ–∫ –¥–ª—è —Å–ª–µ–¥—É—é—â–µ–π –∫–∞—Ä—Ç—ã
            elif line:
                current_map.append(line)  # –î–æ–±–∞–≤–ª—è–µ–º —Å—Ç—Ä–æ–∫—É –≤ —Ç–µ–∫—É—â—É—é –∫–∞—Ä—Ç—É
        if current_map:
            maps.append(current_map)  # –î–æ–±–∞–≤–ª—è–µ–º –ø–æ—Å–ª–µ–¥–Ω—é—é –∫–∞—Ä—Ç—É, –µ—Å–ª–∏ —Ñ–∞–π–ª –Ω–µ –∑–∞–∫–∞–Ω—á–∏–≤–∞–µ—Ç—Å—è –ø—É—Å—Ç–æ–π —Å—Ç—Ä–æ–∫–æ–π
